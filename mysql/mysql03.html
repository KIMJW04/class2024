show databases;

+--------------------+
| information_schema |
| mysql              |
| performance_schema |

데이터베이스 만들기

create database 이름; 
mysql> create database test; //데이터베이스를 test라는 이름으로 만든다
Query OK, 1 row affected (0.00 sec)

mysql> show databases;  //데이터베이스 목록을 보여준다
+--------------------+
| Database           |
+--------------------+
| information_schema |
| mysql              |
| performance_schema |
| sys                |
| test3              |
+--------------------+
5 rows in set (0.00 sec)

데이터베이스 삭제

drop database 이름; 
mysql> drop database test;   //test라는 데이터베이스를 완전히 없애버린다
Query OK, 1 rows affected (0.01 sec)

mysql> show databases;  //데이터베이스 목록을 보여준다
+--------------------+
| Database           |
+--------------------+
| information_schema |
| mysql              |
| performance_schema |
| sys                |
+--------------------+
4 rows in set (0.00 sec)
