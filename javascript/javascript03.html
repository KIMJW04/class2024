10. 화살표 함수 : 매개변수 + 리턴값 함수

화살표 함수를 사용하여 매개변수와 값을 반환하는 예시입니다. 화살표 함수는 (매개변수) => {} 형태로 사용됩니다. 이 예시에서는 매개변수가 있는 함수를 정의하고, 해당 함수에서는 매개변수를 받아서 값을 반환하는 형태로 구성되었습니다. 

{
    // 매개변수와 값을 반환하는 화살표 함수
    const func = (str) => {
        return str;
    }
    console.log(func("10. 함수가 실행되었습니다"));

    // 매개변수가 하나인 경우 괄호 생략 가능
    const func2 = str => {
        return str;
    }
    
    console.log(func2("10. 함수가 실행되었습니다"));
    // 리턴만 하는 경우 중괄호와 return 생략 가능
    const func3 = str => str;
    console.log(func3("10. 함수가 실행되었습니다"));

    // 선언적 함수 (가독성을 위해 괄호 생략하지 않음)
    func4 = str => str;
    console.log(func4("10. 함수가 실행되었습니다"));
}

20. 함수 : 즉시실행 함수

이 예시에서는 자바스크립트의 즉시실행 함수를 소개합니다. 즉시실행 함수는 함수를 정의함과 동시에 즉시 실행되는 형태의 함수입니다. 

{
    // 즉시실행 함수 (Immediately Invoked Function Expression, IIFE)
    (function () {
        console.log("20-1. 함수가 실행되었습니다.");
    })();

    // 화살표 함수를 사용한 즉시실행 함수
    (() => {
        console.log("20-2. 함수가 실행되었습니다.");
    })();
}
결과 확인하기
20-1. 함수가 실행되었습니다.
20-2. 함수가 실행되었습니다.

21. 함수 : 파라미터 함수

이 예시에서는 함수의 매개변수를 활용하여 기본값을 설정하고 함수를 호출하는 방법을 보여줍니다. 

{
    // 기본적인 함수 정의와 매개변수 사용
    function func1(str = "21-1. 실행되었습니다.") {
        console.log(str);
    }
    func1();

    // 화살표 함수와 매개변수 사용
    const func2 = (str = "21-2. 실행되었습니다.") => {
        console.log(str);
    }
    func2();
}
결과 확인하기
21-1. 실행되었습니다.
21-2. 실행되었습니다.

22. 함수 : 재귀 함수

이 예시에서는 반복문과 재귀 함수를 사용하여 함수를 반복적으로 호출하는 방법을 비교합니다. 

{
    // 반복문을 사용한 함수
    function func1(num) {
        for (let i = 0; i < num; i++) {
            console.log("22-1. 실행되었습니다.");
        }
    }
    func1(10);

    // 재귀 함수를 사용한 함수
    function func2(num) {
        if (num < 1) return;
        console.log("22-2. 실행되었습니다.");
        func2(num - 1); // 재귀 호출
    }
    func2(10);
}
결과 확인하기
22-1. 실행되었습니다. (총 10번 출력)
22-2. 실행되었습니다. (총 10번 출력)

23. 함수 : 콜백 함수

이 예시에서는 콜백 함수의 활용 방법을 보여줍니다. 콜백 함수는 다른 함수의 인자로 전달되어 특정 시점에 실행되는 함수입니다. 

{
    // 이벤트 핸들러 등에 사용되는 콜백 함수
    function func1() {
        console.log("23-1. 실행되었습니다.");
    }
    // 예시: 이벤트 리스너에 콜백 함수 등록
    // btn.addEventListener("click", func1);

    // 다른 함수에 인자로 전달되는 콜백 함수
    function func2() {
        console.log("23-2. 실행되었습니다.");
    }
    function func3(callback) {
        callback();
    }
    func3(func2);

    // 반복문을 사용하여 콜백 함수 호출
    function func4(num) {
        console.log("23-" + num + ". 실행되었습니다.");
    }
    function func5(callback) {
        for (let i = 3; i < 5; i++) {
            callback(i);
        }
    }
    func5(func4);
}
결과 확인하기
23-3. 실행되었습니다.
23-4. 실행되었습니다.

24. 함수 : 비동기 방식 : 콜백 함수

이 예시에서는 비동기 방식에서 콜백 함수를 사용하는 방법을 소개합니다. 

{
    // 동기적인 함수 호출
    function func1() {
        console.log("24-1. 실행되었습니다.");
    }
    function func2() {
        console.log("24-2. 실행되었습니다.");
    }
    func1();
    func2();

    // 비동기적인 함수 호출 (주석 해제 후 실행)
     function func3() {
         setTimeout(() => {
             console.log("24-3. 실행되었습니다.");
         }, 1000)
     }
     function func4() {
         console.log("24-4. 실행되었습니다.");
     }
     func3();
     func4();

    // 콜백 함수를 이용한 비동기 호출 (주석 해제 후 실행)
     function func5(callback) {
         setTimeout(() => {
             console.log("24-5. 실행되었습니다.");
             callback();
         }, 1000)
     }
     function func6() {
         console.log("24-6. 실행되었습니다.");
     }
     func5(function () {
         func6();
     });
}
결과 확인하기
24-1. 실행되었습니다.
24-2. 실행되었습니다.
24-4. 실행되었습니다. // func4 함수가 먼저 실행됨
24-3. 실행되었습니다. // func3 함수가 1초 후에 실행됨
24-5. 실행되었습니다. // func5 함수 실행 후 1초 후에 로그 출력
24-6. 실행되었습니다. // func6 함수는 func5 내부 콜백 함수에서 호출됨

26. 함수 : 비동기 방식 : async/await

이 예시에서는 async/await를 사용하여 비동기 함수를 선언하고 호출하는 방법을 보여줍니다. 

{
    // 동기 방식의 함수 선언과 호출
    function func1() {
        console.log("26-1. 실행되었습니다.");
    }
    func1();

    // 비동기 함수 선언과 호출 (async/await)
    async function func2() {
        console.log("26-2. 실행되었습니다.");
    }
    func2();

    // 비동기 함수에서 외부 데이터 받아오기(fetch)
    async function func3() {
        try {
            const result = await fetch("https://example.com/api/data");
            const data = await result.json();
            console.log(data);
        } catch (error) {
            console.log(error);
        }
    }
    func3();
}
결과 확인하기
26-1. 실행되었습니다.
26-2. 실행되었습니다.
(외부 데이터 받아오기 결과 출력)
