01. 1부터 100까지 출력하기(for문, while, do while)

{
    console.log("for")
    for (let i = 1; i <= 100; i++) {
        console.log(i);
    }                       //1,2,3,4,5,6...100
        
    console.log("while");
    let j = 1;
    while (j <= 100) {
        console.log(j);
        j++;
    }                       //1,2,3,4,5,6...100
        
    console.log("do while");
    let x = 1;
    do {
        console.log(x);
        x++
    } while (x <= 100)    //1,2,3,4,5,6...100
}

02. 0부터 100까지 출력하기(for문, if문)

{
    for (let i = 0; i <= 100; i++) {
        if (i % 2 == 0) {
            console.log(i);
        }
    }
}
